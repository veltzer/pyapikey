[docs]class TempStore:
    def __init__(self):
        filename = os.path.expanduser("~/.config/pyapikey.temp.json")
        if os.path.isfile(filename):
            with open(filename) as file_handle:
                self.data = json.load(file_handle, object_hook=object_hook)
        else:
            self.data = {}
